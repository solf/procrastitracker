# Example exclusions file for ProcrastiTracker
# Copy relevant lines to %APPDATA%\procrastitrackerdbs\exclusions.txt
# Or use the Advanced Settings dialog to configure
#
# IMPORTANT NOTES:
# - Exe names are lowercase without .exe extension (e.g., "pwsh" not "pwsh.exe")
# - Patterns match the ENTIRE string (not substrings) - use .* for wildcards
# - Patterns are case-insensitive regex

# Exclude PowerShell exactly
exe:pwsh

# Exclude any window with "password" anywhere in the title
title:.*password.*

# Exclude specific password-related window (e.g., sqlf_pwdWindowTitle from screenshot)
title:sqlf_pwdw.*

# Exclude Windows Terminal exactly
exe:windowsterminal

# Exclude Command Prompt exactly
exe:cmd

# Exclude Notepad exactly
exe:notepad

# Exclude any browser private/incognito windows
title:.*(private|incognito|inprivate).*

# Exclude window titles containing banking keywords
title:.*(bank|paypal|wallet|credit card).*

# Exclude multiple password managers (matches any of them)
exe:(keepass|lastpass|1password|bitwarden)

# Exclude multiple system tools
exe:(taskmgr|regedit|services)

# Exclude all browsers
exe:(chrome|firefox|msedge|opera)

# Exclude window titles that start with "Private -"
title:Private -.*

# Exclude exact window title match
title:My Secret Project
